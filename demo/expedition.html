do Notation
// Block expression — last value is result
fn fleet_stats(missions) -> any {
  do {
    let n        = count(missions)
    let ore      = missions.sum_by(m => m.ore)
    let fuel     = missions.sum_by(m => m.fuel)
    let max_dist = missions.max_by(m => m.distance).distance
    // object shorthand at the end
    { count: n, ore, fuel, max_dist }
  }
}
